def create_graph_from_df(data) -> nx.Graph:
    return nx.from_pandas_adjacency(data)

# Part C
def calc_degrees_for_nodes(G: nx.Graph) -> np.ndarray:
    return np.array([G.degree(node) for node in G.nodes()])

# Part D
def calc_diameter(G: nx.Graph) -> int:
    return nx.diameter(G)

# Part E
def calc_clustering_coefficient(G: nx.Graph) -> float:
    return nx.average_clustering(G)

# Part F
def calc_shortest_paths(G: nx.Graph) -> pd.DataFrame:
    shortestpathdata = pd.DataFrame(columns=['Source', 'Target', 'ShortestPath'])
    for source in G.nodes():
        for dest in G.nodes():
            # compute shortest path
            try :
                shortest_path_length = nx.shortest_path_length(G, source=source, target=dest)
                
            except nx.NetworkXNoPath:
                shortest_path_length = -1

            finally: 
                shortestpathdata.loc[len(shortestpathdata)] = {'Source': source, 'Target': dest, 'ShortestPath': shortest_path_length}
    return shortestpathdata

def plot_shortest_paths(G: nx.Graph):
    shortestpathdata = calc_shortest_paths(G)
    plt.hist(shortestpathdata['ShortestPath'])
    plt.xlabel('Shortest Path Length')
    plt.ylabel('Frequency')
    plt.title('Shortest Path Length Distribution')
    plt.show()
    return shortestpathdata

# Part G
def plot_degree_distribution(G: nx.Graph):
    degrees = calc_degrees_for_nodes(G)
    # plot histogram of degrees
    plt.hist(degrees)
    plt.xlabel('Degree')
    plt.ylabel('Frequency')
    plt.title('Degree Distribution')
    plt.show()
    
    return degrees

# Part H
def calc_betweenness_centrality(G: nx.Graph) -> pd.DataFrame:
    betweenness_centrality = nx.betweenness_centrality(G)
    betweenness_centrality_df = pd.DataFrame(betweenness_centrality.items(), columns=['Node', 'BetweennessCentrality'])
    return betweenness_centrality_df.sort_values(by='BetweennessCentrality', ascending=False)

def plot_betweenness_centrality(G: nx.Graph):
    betweenness_centrality_df = calc_betweenness_centrality(G)
    plt.hist(betweenness_centrality_df['BetweennessCentrality'])
    plt.xlabel('Betweenness Centrality')
    plt.ylabel('Frequency')
    plt.title('Betweenness Centrality Distribution')
    plt.show()
    return betweenness_centrality_df

def calc_local_clustering_coefficient(G: nx.Graph) -> pd.DataFrame:

    return local_clustering_coefficient_df.sort_values(by='LocalClusteringCoefficient', ascending=False)

    plt.show()
    return local_clustering_coefficient_df
